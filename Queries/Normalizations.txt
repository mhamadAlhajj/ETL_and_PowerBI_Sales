create products table
------------------------

Create table products (
ProductId int primary key not null,
ProductName nvarchar(100),
Cat_SubCategoryId int ,
BrandId int ,
UnitPrice float
Constraint FK_Category Foreign key (Cat_SubCategoryId) references [dbo].[Cat_SubCategories](Cat_SubCategories_Id)  on Delete cascade ,
Constraint FK_Brand Foreign key (BrandId) references [dbo].[brands](BrandId) on Delete cascade
)

create table orders
-------------------
CREATE TABLE [dbo].[Orders](
	[OrderId] [int] NOT NULL,
	[OrderDate] [date] NULL,
	[CustomerId] [int] NULL,
	[SalesRepId] [int] NULL,
	[PaymentMethod] [nvarchar](30) NULL,
constraint FK_CustomerOrders foreign key (CustomerId) references [dbo].[Customers] ([CustomerId]) on delete cascade ,
constraint FK_RepresentativeOrders foreign key (SalesRepId) references [dbo].[Representative] ([RepresentativeId]) on delete cascade 
)

create table orderItem
------------------------
create table orderItems(
OrderItem_Id int primary key not null ,
ProductId int ,

Quantity int ,
discount float ,
constraint FK_ProductItem foreign key (ProductId) references [dbo].[products](ProductId) on delete Cascade,
